# Toast

export function createToast({
                              type,
                              message,
                              heading
                            }) {
  const toaster = document.querySelector('studs-toaster');
  toaster.createToast({
    type: type || 'info',
    message: message || 'This is an info toast',
    heading: heading || ''
  })
}
